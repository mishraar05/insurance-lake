```python
from databricks.sdk import WorkspaceClient
from databricks.sdk.service.jobs import (
    Task, NotebookTask, JobCluster, ClusterSpec
)

class FrameworkJobCreator:
    """Creates test jobs for generated frameworks"""
    
    def __init__(self, workspace_client: WorkspaceClient):
        self.w = workspace_client
    
    def create_test_job(
        self,
        framework_name: str,
        notebook_path: str,
        cluster_config: dict
    ) -> dict:
        """
        Create a test job for the generated framework
        
        Returns:
            {
                "job_id": 12345,
                "job_url": "https://...",
                "run_id": 67890
            }
        """
        job = self.w.jobs.create(
            name=f"[TEST] {framework_name}",
            tasks=[
                Task(
                    task_key="test_framework",
                    notebook_task=NotebookTask(
                        notebook_path=notebook_path,
                        source="WORKSPACE"
                    ),
                    job_cluster_key="test_cluster"
                )
            ],
            job_clusters=[
                JobCluster(
                    job_cluster_key="test_cluster",
                    new_cluster=ClusterSpec(**cluster_config)
                )
            ],
            tags={"generated_by": "framework_builder_bot"}
        )
        
        # Trigger initial run
        run = self.w.jobs.run_now(job_id=job.job_id)
        
        return {
            "job_id": job.job_id,
            "job_url": f"{self.w.config.host}/#job/{job.job_id}",
            "run_id": run.run_id
        }
```

### 5. SOLID Principles & Design Patterns ✅

**Goal:** Design for extensibility, implement standard first, enable multi-customer customization

#### A. SOLID Principles Application

**1. Single Responsibility Principle (SRP)**

Each class has ONE reason to change:

```python
# BAD: God class with multiple responsibilities
class IngestionPipeline:
    def read_data(self): pass
    def validate_schema(self): pass
    def transform_data(self): pass
    def write_to_delta(self): pass
    def send_notifications(self): pass

# GOOD: Separate responsibilities
class Reader:
    def read(self, path: str) -> DataFrame: pass

class SchemaValidator:
    def validate(self, df: DataFrame) -> bool: pass

class Transformer:
    def transform(self, df: DataFrame) -> DataFrame: pass

class Writer:
    def write(self, df: DataFrame, table: str): pass

class NotificationService:
    def send(self, message: str): pass
```

**2. Open/Closed Principle (OCP)**

Open for extension, closed for modification:

```python
# Base class (closed for modification)
class LoadStrategy(ABC):
    @abstractmethod
    def write(self, df: DataFrame, target: str) -> dict:
        pass

# Extensions (open for extension)
class AppendStrategy(LoadStrategy):
    def write(self, df: DataFrame, target: str) -> dict:
        df.write.mode("append").saveAsTable(target)
        return {"rows_written": df.count()}

class SCD2Strategy(LoadStrategy):
    def write(self, df: DataFrame, target: str) -> dict:
        # SCD2 logic here
        pass
```

**3. Liskov Substitution Principle (LSP)**

Subclasses are substitutable for their base class:

```python
def process_data(reader: Reader, strategy: LoadStrategy):
    """Works with ANY reader and ANY strategy"""
    df = reader.read(path)
    result = strategy.write(df, target_table)
    return result

# All of these work:
process_data(CSVReader(), AppendStrategy())
process_data(JSONReader(), SCD2Strategy())
process_data(DeltaReader(), FullRefreshStrategy())
```

**4. Interface Segregation Principle (ISP)**

Many client-specific interfaces > one general interface:

```python
# BAD: Fat interface
class DataSource(ABC):
    @abstractmethod
    def read_batch(self): pass
    @abstractmethod
    def read_stream(self): pass  # Not all sources support streaming!
    @abstractmethod
    def write(self): pass  # Sources don't write!

# GOOD: Segregated interfaces
class Reader(ABC):
    @abstractmethod
    def read(self) -> DataFrame: pass

class StreamingReader(Reader):
    @abstractmethod
    def read_stream(self) -> DataFrame: pass

class Writer(ABC):
    @abstractmethod
    def write(self, df: DataFrame): pass
```

**5. Dependency Inversion Principle (DIP)**

Depend on abstractions, not concretions:

```python
# BAD: Depends on concrete implementation
class IngestionEngine:
    def __init__(self):
        self.reader = CSVReader()  # Tightly coupled!
        self.strategy = SCD2Strategy()  # Tightly coupled!

# GOOD: Depends on abstractions
class IngestionEngine:
    def __init__(
        self,
        reader: Reader,  # Abstract interface
        strategy: LoadStrategy  # Abstract interface
    ):
        self.reader = reader
        self.strategy = strategy
```

#### B. Design Patterns

**1. Strategy Pattern** (for load strategies)

```python
# Strategy interface
class LoadStrategy(ABC):
    @abstractmethod
    def write(self, df: DataFrame, target: str) -> dict:
        pass

# Concrete strategies
class AppendStrategy(LoadStrategy): pass
class SCD1Strategy(LoadStrategy): pass
class SCD2Strategy(LoadStrategy): pass
class FullRefreshStrategy(LoadStrategy): pass

# Context uses strategy
class IngestionEngine:
    def __init__(self, strategy: LoadStrategy):
        self._strategy = strategy
    
    def set_strategy(self, strategy: LoadStrategy):
        """Runtime strategy switching"""
        self._strategy = strategy
    
    def ingest(self, df: DataFrame, target: str):
        return self._strategy.write(df, target)
```

**2. Factory Pattern** (for reader creation)

```python
# Factory
class ReaderFactory:
    _readers: dict[str, type[Reader]] = {}
    
    @classmethod
    def register(cls, format: str, reader_class: type[Reader]):
        cls._readers[format] = reader_class
    
    @classmethod
    def get_reader(cls, format: str) -> Reader:
        if format not in cls._readers:
            raise ValueError(f"No reader for format: {format}")
        return cls._readers[format]()

# Register readers
ReaderFactory.register("csv", CSVReader)
ReaderFactory.register("json", JSONReader)
ReaderFactory.register("parquet", ParquetReader)

# Usage
reader = ReaderFactory.get_reader("csv")
df = reader.read(path)
```

**3. Template Method Pattern** (for engine workflows)

```python
class IngestionEngine(ABC):
    """Template for ingestion workflow"""
    
    def ingest(self, feed_config: dict) -> dict:
        """Template method (defines workflow)"""
        
        # 1. Pre-processing hooks
        self._before_read(feed_config)
        
        # 2. Read data (implemented by subclass)
        df = self.read_data(feed_config)
        
        # 3. Validate (common logic)
        self._validate_schema(df, feed_config)
        
        # 4. Transform (implemented by subclass)
        df = self.transform_data(df, feed_config)
        
        # 5. Write (implemented by subclass)
        result = self.write_data(df, feed_config)
        
        # 6. Post-processing hooks
        self._after_write(result)
        
        return result
    
    # Template steps (to be implemented by subclasses)
    @abstractmethod
    def read_data(self, config: dict) -> DataFrame: pass
    
    @abstractmethod
    def transform_data(self, df: DataFrame, config: dict) -> DataFrame: pass
    
    @abstractmethod
    def write_data(self, df: DataFrame, config: dict) -> dict: pass
    
    # Common logic (shared by all subclasses)
    def _validate_schema(self, df: DataFrame, config: dict): pass
    def _before_read(self, config: dict): pass
    def _after_write(self, result: dict): pass
```

**4. Builder Pattern** (for complex configuration)

```python
class IngestionConfigBuilder:
    """Builder for IngestionConfig"""
    
    def __init__(self):
        self._config = {}
    
    def with_source(self, format: str, path: str):
        self._config["source"] = {"format": format, "path": path}
        return self
    
    def with_strategy(self, strategy: str, **options):
        self._config["strategy"] = {"type": strategy, "options": options}
        return self
    
    def with_target(self, table: str):
        self._config["target"] = table
        return self
    
    def build(self) -> dict:
        # Validation logic
        if "source" not in self._config:
            raise ValueError("Source is required")
        return self._config

# Usage
config = (
    IngestionConfigBuilder()
    .with_source(format="csv", path="/data/claims.csv")
    .with_strategy("scd2", primary_keys=["claim_id"])
    .with_target("silver.claims")
    .build()
)
```

**5. Dependency Injection** (for testability and flexibility)

```python
# BAD: Hard-coded dependencies
class IngestionEngine:
    def __init__(self):
        self.config_loader = ConfigLoader()  # Hard-coded
        self.abc_sdk = ABC()  # Hard-coded

# GOOD: Injected dependencies
class IngestionEngine:
    def __init__(
        self,
        config_loader: ConfigLoader,
        abc_sdk: ABC,
        reader: Reader,
        strategy: LoadStrategy
    ):
        self._config_loader = config_loader
        self._abc = abc_sdk
        self._reader = reader
        self._strategy = strategy

# Usage (with DI container)
engine = IngestionEngine(
    config_loader=ConfigLoader(),
    abc_sdk=ABC(),
    reader=ReaderFactory.get_reader("csv"),
    strategy=SCD2Strategy()
)
```

#### C. Multi-Customer Extensibility Design

**Goal:** One codebase, multiple customer configurations

**Pattern: Strategy + Config-Driven**

```python
# Core framework (shared)
class IngestionEngine:
    def __init__(self, customer_config: CustomerConfig):
        self._config = customer_config
        self._reader = self._config.get_reader()
        self._strategy = self._config.get_strategy()
    
    def ingest(self, feed_id: str):
        feed = self._config.get_feed(feed_id)
        df = self._reader.read(feed.source_path)
        return self._strategy.write(df, feed.target_table)

# Customer-specific config
class CustomerConfig(ABC):
    @abstractmethod
    def get_reader(self) -> Reader: pass
    
    @abstractmethod
    def get_strategy(self) -> LoadStrategy: pass
    
    @abstractmethod
    def get_feed(self, feed_id: str) -> Feed: pass

# Customer A configuration
class CustomerAConfig(CustomerConfig):
    def get_reader(self) -> Reader:
        return CSVReader(delimiter="|")  # Custom delimiter
    
    def get_strategy(self) -> LoadStrategy:
        return SCD2Strategy()  # SCD2 for Customer A
    
    def get_feed(self, feed_id: str) -> Feed:
        # Customer A's feed metadata
        return Feed.load(f"conf/customers/customer_a/{feed_id}.json")

# Customer B configuration
class CustomerBConfig(CustomerConfig):
    def get_reader(self) -> Reader:
        return JSONReader(multiLine=True)  # Multi-line JSON
    
    def get_strategy(self) -> LoadStrategy:
        return AppendStrategy()  # Append-only for Customer B
    
    def get_feed(self, feed_id: str) -> Feed:
        # Customer B's feed metadata
        return Feed.load(f"conf/customers/customer_b/{feed_id}.json")

# Usage
engine_a = IngestionEngine(CustomerAConfig())
engine_b = IngestionEngine(CustomerBConfig())
```

**Directory Structure for Multi-Customer:**
```

  │   ├── customer_a/

  │   │   └── feeds/

  │   ├── customer_b/
  │   │   ├── config.yaml         # Customer B settings
  │   │   └── feeds/
  │   │       └── policies.json
  │   └── _default/
  │       └── config.yaml         # Default settings
```
